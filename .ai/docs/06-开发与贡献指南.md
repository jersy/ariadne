const fs = require('fs')
const path = require('path')
const glob = require('glob')

function extractFrontmatter(filePath) {
  const content = fs.readFileSync(filePath, 'utf-8')
  const match = content.match(/^---\n([\s\S]*?)\n---/)

  if (!match) {
    throw new Error(`Missing frontmatter: ${filePath}`)
  }

  return match[1]
}

function validateAgentFrontmatter(filePath) {
  const fm = extractFrontmatter(filePath)

  if (!fm.includes('name:')) throw new Error(`Missing name: ${filePath}`)
  if (!fm.includes('description:')) throw new Error(`Missing description: ${filePath}`)
  if (!fm.includes('tools:')) throw new Error(`Missing tools: ${filePath}`)
  if (!fm.includes('model:')) throw new Error(`Missing model: ${filePath}`)
}

function validateSkillFrontmatter(filePath) {
  const fm = extractFrontmatter(filePath)

  if (!fm.includes('name:')) throw new Error(`Missing name: ${filePath}`)
  if (!fm.includes('description:')) throw new Error(`Missing description: ${filePath}`)
}
